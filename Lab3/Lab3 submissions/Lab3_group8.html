Monte_Carlo<- function(n,a,b){
   x <- runif(n,a,b)
  g_of_x <- dnorm(x,1,10)
  Int <- mean(g_of_x)*(b-a)
  return(Int)
}


Monte_Carlo_aprox <- function(n,a,b){
  Int <- Monte_Carlo(n,a,b)
  True_val<- pnorm(b,1,10)-(pnorm(a,1,10))
  Error <- Int-(pnorm(b,1,10)-(pnorm(a,1,10)))
  print(list("The value of monte carlo approximation"=Int,"Error"=Error, "True_val"=True_val))
}

f.stock.return<- function(sp,n,sm,ss){
  eps<- runif(n,0,1)
  predic_stock<- sp*(1 + qnorm(eps, sm*(1/n),ss*(sqrt(1/n))))
  return(c(sp,predic_stock))
}
